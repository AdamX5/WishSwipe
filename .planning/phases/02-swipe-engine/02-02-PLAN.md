useSprings + useDrag canonical pattern (from 02-RESEARCH.md):
```typescript
import { useSprings, animated, to as interpolate } from '@react-spring/web'
import { useDrag } from '@use-gesture/react'

const SWIPE_VELOCITY_THRESHOLD = 0.2  // horizontal velocity (px/ms)
const SWIPE_DISTANCE_THRESHOLD = 100  // px

const to = (i: number) => ({ x: 0, y: i * -4, scale: 1, rot: 0 })
const from = () => ({ x: 0, rot: 0, scale: 1.5, y: -1000 })
const trans = (r: number, s: number) => `rotateZ(${r}deg) scale(${s})`

// Fixed spring count (20) — hooks cannot have dynamic call counts
const [springs, api] = useSprings(20, i => ({ ...to(i), from: from() }))

const bind = useDrag(({ args: [index], active, movement: [mx], direction: [xDir], velocity: [vx], last }) => {
  const trigger = vx > SWIPE_VELOCITY_THRESHOLD || Math.abs(mx) > SWIPE_DISTANCE_THRESHOLD
  const dir = xDir < 0 ? -1 : 1  // -1 = left, +1 = right

  if (last && trigger) {
    gone.current.add(index)
    const direction = dir === 1 ? 'right' : 'left'
    // Call recordSwipe here — ONCE per completed gesture
  }

  api.start(i => {
    if (index !== i) return  // only animate the dragged card
    const isGone = gone.current.has(index)
    return {
      x: isGone ? (200 + window.innerWidth) * dir : active ? mx : 0,
      rot: mx / 100 + (isGone ? dir * 10 * vx : 0),
      scale: active ? 1.1 : 1,
      config: {
        friction: 50,
        tension: active ? 800 : isGone ? 200 : 500,
      },
    }
  })
})
```

const bind = useDrag(({ args: [index], active, movement: [mx], direction: [xDir], velocity: [vx], last }) => {
  const trigger = vx > 0.2 || Math.abs(mx) > 100
  const dir = xDir < 0 ? -1 : 1

  if (last && trigger) {
    gone.current.add(index)
    const swipeDir = dir === 1 ? 'right' : 'left'
    const product = queue[index]

          sourceStore: product.sourceStore,

    const isGone = gone.current.has(index)

**Render — 3-card DOM invariant:**
Show only the top 3 cards from the queue. Slice `queue.slice(0, 3)` for display. Pair each displayed card with its corresponding spring from `springs` by absolute index in queue (index 0 = bottom/Preview, index 2 = top/Active — highest index renders on top via `zIndex`).

```tsx
// Stack: index 0 = bottom (Preview), index 1 = middle (Next), index 2 = top (Active)
// Reverse the array so highest z-index is the current top card
const displayQueue = queue.slice(0, 3)

return (
  <div className="relative flex items-center justify-center" style={{ height: '24rem' }}>
    {queue.length === 0 ? (
      <p className="text-gray-400">No more products — check back later</p>
    ) : (
      displayQueue.map((product, i) => {
        const springIndex = displayQueue.length - 1 - i  // top card gets highest spring index
        const { x, y, rot, scale } = springs[i]
        return (
          <animated.div
            key={product._id}
            style={{ x, y, position: 'absolute', zIndex: displayQueue.length - i }}
          >
            <animated.div
              {...bind(i)}
              style={{ transform: interpolate([rot, scale], trans), touchAction: 'none' }}
            >
              <SwipeCard product={product} xSpring={x} />
            </animated.div>
          </animated.div>
        )
      })
    )}
    {queue.length > 0 && queue.length < 3 && (
      <div
        className="absolute rounded-2xl bg-gray-100"
        style={{ width: '18rem', height: '24rem', zIndex: 0 }}
      />
    )}
  </div>
)
```

Add `touchAction: 'none'` on the draggable animated.div to prevent browser scroll interference on mobile.
